Database SQL Operation: Task_wa.searchParticipantOpenTasks

Operation Details
Description:  
Sterotype: nsmulti

SQL 
SELECT
  Task_wa.taskID,
  Task_wa.subject,
  Task_wa.reservedBy,
  Task_wa.dueDateTime,
  Task_wa.priority,
  Task_wa.status,
  Task_wa.versionNo
INTO
  :taskID,
  :subject,
  :reservedBy,
  :dueDateTime,
  :priority,
  :status,
  :versionNo 
FROM
  Task_wa 
WHERE Task_wa.taskID IN 
  ( 
  SELECT
    Task_wa.taskID 
  FROM
    Task_wa,
    TaskLink 
  WHERE 
    (
      
      (
        TaskLink.linkedID = :linkedID 
      ) 
      AND 
      (
        TaskLink.linkType = :linkedType 
      ) 
      AND 
      (
        TaskLink.taskID = Task_wa.taskID 
      ) 
    ) 
  ) 
  AND 
  (
    
    (
      Task_wa.status <> :status 
    ) 
    OR 
    (
      :status IS NULL 
    ) 
  ) 
ORDER BY Task_wa.taskID

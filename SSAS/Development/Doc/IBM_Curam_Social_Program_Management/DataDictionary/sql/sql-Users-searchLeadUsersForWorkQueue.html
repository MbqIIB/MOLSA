Database SQL Operation: Users.searchLeadUsersForWorkQueue

Operation Details
Description: Method to return a list of supervisors where the case owner is a work queue. This list contains all users holding lead positions, for the organization units of the positions, that are assigned to all the subscribers of the specified work queue.
Sterotype: nsmulti

SQL 
SELECT
  distinct(PositionHolderLink.userName)
INTO
  :userName 
FROM
  Position,
  PositionHolderLink,
  OrgUnitPositionLink 
WHERE PositionHolderLink.organisationStructureID = :organisationStructureID 
  AND PositionHolderLink.recordStatus = :recordStatus 
  AND PositionHolderLink.fromDate <= :effectiveDate 
  AND 
  (
    PositionHolderLink.toDate >= :effectiveDate 
    OR PositionHolderLink.toDate IS NULL 
  ) 
  AND OrgUnitPositionLink.positionID = PositionHolderLink.positionID 
  AND OrgUnitPositionLink.organisationStructureID = :organisationStructureID 
  AND Position.leadPositionInd = '1' 
  AND Position.positionID = PositionHolderLink.positionID 
  AND OrgUnitPositionLink.organisationUnitID IN 
  ( 
  SELECT
    OrgUnitPositionLink.organisationUnitID 
  FROM
    OrgUnitPositionLink 
  WHERE OrgUnitPositionLink.positionID IN 
    ( 
    SELECT
      PositionHolderLink.positionID 
    FROM
      PositionHolderLink 
    WHERE PositionHolderLink.userName IN 
      ( 
      SELECT
        WorkQueueSubscription.userName 
      FROM
        WorkQueueSubscription 
      WHERE WorkQueueSubscription.workQueueID = :workQueueID 
      ) 
    ) 
    AND OrgUnitPositionLink.recordStatus = :recordStatus 
  )

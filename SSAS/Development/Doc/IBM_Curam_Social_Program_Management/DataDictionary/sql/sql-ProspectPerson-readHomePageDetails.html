Database SQL Operation: ProspectPerson.readHomePageDetails

Operation Details
Description: Retrieves the details of the prospect person required to be displayed on the prospect person home page.
Sterotype: ns

SQL 
SELECT
  ProspectPerson.personBirthName,
  ProspectPerson.dateOfBirth,
  ProspectPerson.dateOfDeath,
  ProspectPerson.gender,
  ProspectPerson.specialInterestCode,
  ProspectPerson.maritalStatusCode,
  ProspectPerson.nationalityCode,
  ProspectPerson.motherBirthSurname,
  ProspectPerson.comments,
  ProspectPerson.dateOfBirthVerInd,
  ProspectPerson.dateOfDeathVerInd,
  ProspectPerson.fromAge,
  ProspectPerson.toAge,
  ProspectPerson.countryOfBirth,
  ProspectPerson.placeOfBirth,
  ProspectPerson.ethnicOriginCode,
  ProspectPerson.concernRoleID,
  ProspectPerson.personConcernRoleID,
  ProspectPerson.race,
  ProspectPerson.indigenousGroupCode,
  ConcernRole.primaryAlternateID,
  ConcernRole.registrationDate,
  ConcernRole.statusCode,
  ConcernRole.prefPublicOfficeID,
  ConcernRole.preferredLanguage,
  ConcernRole.sensitivity,
  ConcernRole.prefCommMethod,
  ConcernRole.prefCommFromDate,
  ConcernRole.prefCommToDate,
  ConcernRole.primaryAddressID,
  ConcernRole.primaryPhoneNumberID,
  ConcernRole.concernRoleName,
  ConcernRole.concernRoleType,
  AlternateName.title,
  AlternateName.firstForename,
  AlternateName.otherForename,
  AlternateName.surname,
  AlternateName.nameSuffix,
  AlternateName.initials
INTO
  :personBirthName,
  :dateOfBirth,
  :dateOfDeath,
  :gender,
  :specialInterestCode,
  :maritalStatusCode,
  :nationalityCode,
  :motherBirthSurname,
  :comments,
  :dateOfBirthVerInd,
  :dateOfDeathVerInd,
  :fromAge,
  :toAge,
  :countryOfBirth,
  :placeOfBirth,
  :ethnicOriginCode,
  :concernRoleID,
  :personConcernRoleID,
  :race,
  :indigenousGroupCode,
  :primaryAlternateID,
  :registrationDate,
  :statusCode,
  :preferredPublicOfficeID,
  :preferredLanguage,
  :sensitivity,
  :prefCommMethod,
  :prefCommFromDate,
  :prefCommToDate,
  :primaryAddressID,
  :primaryPhoneNumberID,
  :fullName,
  :concernRoleType,
  :title,
  :firstForename,
  :otherForename,
  :surname,
  :nameSuffix,
  :initials 
FROM
  ProspectPerson,
  ConcernRole,
  AlternateName 
WHERE ProspectPerson.concernRoleID = :concernRoleID 
  AND ConcernRole.concernRoleID = :concernRoleID 
  AND AlternateName.alternateNameID = ProspectPerson.primaryAlternateNameID

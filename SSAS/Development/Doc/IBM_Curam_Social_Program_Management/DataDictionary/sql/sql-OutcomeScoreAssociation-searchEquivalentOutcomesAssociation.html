Database SQL Operation: OutcomeScoreAssociation.searchEquivalentOutcomesAssociation

Operation Details
Description:  
Sterotype: nsmulti

SQL 
SELECT
  minimumScore,
  maximumScore,
  flag
INTO
  :minimumScore,
  :maximumScore,
  :flag 
FROM
  OutcomeScoreAssociation 
WHERE decisionMatrixVersionID = :decisionMatrixVersionID 
  AND 
  (
    
    (
      :flag = 4 
      AND 
      (
        
        (
          :minimumScore BETWEEN minimumScore AND maximumScore 
        ) 
        OR 
        (
          :maximumScore BETWEEN minimumScore AND maximumScore 
        ) 
        OR 
        (
          minimumScore IS NULL 
          AND maximumScore >= :minimumScore 
        ) 
        OR 
        (
          maximumScore IS NULL 
          AND minimumScore <= :maximumScore 
        ) 
      ) 
    ) 
    OR 
    (
      :flag = 2 
      AND 
      (
        flag = 2 
        OR 
        (
          :minimumScore <= maximumScore 
        ) 
      ) 
    ) 
    OR 
    (
      :flag = 3 
      AND 
      (
        flag = 3 
        OR 
        (
          :maximumScore >= minimumScore 
        ) 
      ) 
    ) 
  )

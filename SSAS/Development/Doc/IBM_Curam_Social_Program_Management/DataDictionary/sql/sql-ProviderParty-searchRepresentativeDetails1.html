Database SQL Operation: ProviderParty.searchRepresentativeDetails1

Operation Details
Description: Retrieves the list of representatives that satisfies the input search criteria.
Sterotype: nsmulti

SQL 
SELECT
  DISTINCT(ConcernRole.concernRoleID),
  ConcernRole.concernRoleName,
  ConcernRole.registrationDate ,
  ConcernRole.statusCode,
  ConcernRole.primaryAlternateID,
  city.elementValue,
  addressLine1.elementValue
INTO
  :concernRoleID,
  :concernRoleName,
  :startDate,
  :status,
  :referenceNo,
  :city,
  :addressLine1 
FROM
  Representative,
  ConcernRole 
  
  LEFT OUTER JOIN
  
  ProviderParty 
  ON ConcernRole.concernRoleID = ProviderParty.partyConcernRoleID 
  
  LEFT OUTER JOIN
  
  TrainingProgramMember 
  ON ConcernRole.concernRoleID = TrainingProgramMember.partyConcernRoleID,
  AddressElement city,
  AddressElement addressLine1 
WHERE 
  (
    :searchByReferenceNo = '0' 
    OR ConcernRole.primaryAlternateID = :referenceNo 
  ) 
  AND addressLine1.addressID = ConcernRole.primaryAddressID 
  AND addressLine1.elementType = :addressLine1Type 
  AND city.addressID = ConcernRole.primaryAddressID 
  AND city.elementType = :cityType 
  AND 
  (
    :searchByAddrLine = '0' 
    OR addressLine1.upperElementValue like :addressLine1 
  ) 
  AND 
  (
    :searchByFirstName = '0' 
    OR 
    (
      Representative.concernRoleID = ConcernRole.concernRoleID 
      AND UPPER(Representative.representativeName) LIKE :firstName 
    ) 
  ) 
  AND 
  (
    ConcernRole.concernRoleType = :concernRoleType 
    AND ProviderParty.partyConcernRoleID = ConcernRole.concernRoleID 
    AND ProviderParty.recordStatus = :providerPartyStatus 
  ) 
  AND 
  (
    :providerMembershipPresent = '0' 
    OR 
    (
      ProviderParty.category = :providerPartyCategory 
      AND ProviderParty.recordStatus = :providerPartyStatus 
    ) 
  ) 
  AND 
  (
    :personWithTraining = '0' 
    OR 
    (
      TrainingProgramMember.partyConcernRoleID = ConcernRole.concernRoleID 
      AND TrainingProgramMember.status <> :trainingProgramStatus 
    ) 
  )

Database SQL Operation: DecisionMatrixVersion.searchByDeterminationPackageVersionKey

Operation Details
Description:  
Sterotype: nsmulti

SQL 
SELECT
  DMV1.decisionMatrixVersionID,
  DMV1.name,
  DMV1.type,
  DMV1.decisionMatrixVersionNumber,
  DMV1.versionStatus,
  DMV1.decisionMatrixID,
  DMV1.fromDate,
  DMV1.toDate
INTO
  :decisionMatrixVersionID,
  :name,
  :type,
  :versionNumber,
  :versionStatus,
  :decisionMatrixID,
  :fromDate,
  :toDate 
FROM
  DecisionMatrixVersion DMV1,
  DPDecisionMatrixLink DPL1 
WHERE DMV1.decisionMatrixVersionID = DPL1.decisionMatrixVersionID 
  AND DPL1.determinationPackageVersionID = :determinationPackageVersionID 
  AND decisionMatrixVersionNumber = 
  ( 
  SELECT
    max(decisionMatrixVersionNumber) 
  FROM
    DecisionMatrixVersion DMV2,
    DPDecisionMatrixLink DPL2 
  WHERE DMV2.decisionMatrixVersionID = DPL2.decisionMatrixVersionID 
    AND DPL1.determinationPackageVersionID = DPL2.determinationPackageVersionID 
    AND DMV1.decisionMatrixID = DMV2.decisionMatrixID 
  ) 
ORDER BY DMV1.name

Example:

Technical Notes:

SQL 
select
   t2.CONCERNROLEID as CONCERNROLEID,
   t2.DWADDRESSID as DWADDRESSID,
   t2.REGISTRATIONDATE as REGDATE,
   t3.INDUSTRYTYPE as INDUSTRYTYPE,
   t1.DWPARTICIPANTSTATUSID as DWPARTICIPANTSTATUSID,
   t1.STARTDATE as STARTDATE,
   t1.ENDDATE as ENDDATE,
   T1.LASTWRITTEN as LASTWRITTEN,
   T3.DWINDUSTRYKEY as DWINDUSTRYKEY
from
   DW_PARTICIPANTHISTORY t1,
   DW_PARTICIPANT t2,
   DW_EMPLOYER t3
where
   t2.DWPARTICIPANTID = t1.DWPARTICIPANTID
   and t2.DWPARTICIPANTID = t3.dwparticipantid;

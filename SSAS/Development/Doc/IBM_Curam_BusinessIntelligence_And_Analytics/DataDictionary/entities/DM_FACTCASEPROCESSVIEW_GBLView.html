SQL 
select
   coalesce(t4.DWORGUNITID, t8.DWORGUNITID) as DWORGUNITID,
   t3.DWPRODUCTID,
   t2.dwprogramid,
   coalesce(t4.CASEOWNERID, substr(t6.name, 1, 30)) as CASEOWNERID,
   t1.DWSTATUSID,
   t1.LASTWRITTEN,
   t1.FROMDATE,
   t1.TODATE,
   t2.CASEID,
   t1.DESCRIPTION,
   coalesce(t5.DWADDRESSID, -1) as DWADDRESSID,
   T2.CASEREFERENCE,
   T2.DWINTEGRATEDCASETYPEKEY,
   T2.DWCASETYPEKEY
from
   DW_CASE t2
inner join
   DW_CASESTATUSHISTORY t1 on t1.DWCASEID = t2.DWCASEID
inner join
   DW_PRODUCT t3 on t2.DWPRODUCTID = t3.DWPRODUCTID
inner join 
   DW_CASEOWNER t4 on t2.DWCASEOWNERID = t4.DWCASEOWNERID
left outer join 
   DW_ORGUNIT t6 on t4.DWORGUNITID = t6.DWORGUNITID
left outer join 
   DW_USERS t8 on t4.CASEOWNERID = t8.CASEOWNERID
left outer join    
   DM_CLIENTPRIMARYADDRESS_GBL t5 on t2.dwcaseid = t5.dwcaseid;

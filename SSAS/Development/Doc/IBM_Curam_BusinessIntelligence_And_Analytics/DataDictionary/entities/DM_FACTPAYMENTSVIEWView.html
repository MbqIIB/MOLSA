Description: DM_FACTPAYMENTSVIEW : Fact Payments View

Description:

This view returns the following attributes:

Example:

Technical Notes:

SQL 
select    dw_paymenthistory.dwfinancialstatusid,
          dw_payment.dwdeductionid,
          dw_payment.dwtransactionid, 
          dw_payment.dwpersonhistoryid,
          dw_payment.instructlineitemid,
          dw_payment.gaurdianflag, 
          dw_payment.regeneratedflag,
          dw_payment.amount,
          dw_payment.unprocessedamount, 
          dw_case.caseid,
          dw_case.dwprogramid,
          coalesce(dw_caseowner.dworgunitid, DW_USERS.DWORGUNITID) as dworgunitid,
          dw_case.dwproductid, 
          dw_participant.concernroleid,
          dw_payment.lastwritten,
          dw_payment.deliverymethod, 
          dw_paymenthistory.startdate 
from      dw_financialstatus, 
          dw_paymenthistory, 
          dw_payment, 
          dw_case, 
          dw_participant, 
          dw_caseowner,
          DW_USERS
where     (DW_PAYMENTHISTORY.ENDDATE IS NULL 
            OR (DW_PAYMENTHISTORY.ENDDATE IS NOT NULL 
              AND  DW_FINANCIALSTATUS.STATUSCODE = 'SUS')) 
AND       DW_PAYMENTHISTORY.DWPAYMENTID  =  DW_PAYMENT.DWPAYMENTID 
AND       DW_PAYMENT.DWCASEID  =  DW_CASE.DWCASEID 
AND       DW_PAYMENT.DWPARTICIPANTID  =  DW_PARTICIPANT.DWPARTICIPANTID 
AND       DW_CASEOWNER.DWCASEOWNERID  =  DW_CASE.DWCASEOWNERID 
AND       DW_FINANCIALSTATUS.DWFINANCIALSTATUSID = DW_PAYMENTHISTORY.DWFINANCIALSTATUSID
AND       dw_caseowner.CASEOWNERID = DW_USERS.CASEOWNERID (+);

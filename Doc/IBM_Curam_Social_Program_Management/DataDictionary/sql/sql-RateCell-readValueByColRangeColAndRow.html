Database SQL Operation: RateCell.readValueByColRangeColAndRow

Operation Details
Description: Read a cell value by rate header, effective date, column type, column range, row type, and value.
Sterotype: ns

SQL 
SELECT
  rcValue.rateCellValue,
  rcValue.rateCellID
INTO
  :rateCellValue,
  :rateCellID 
FROM
  RateCell rcValue,
  RateCell rcRange,
  RateColumn rcolRange,
  RateColumn rCol,
  RateRow,
  RateHeader 
WHERE effectiveDate = 
  ( 
  SELECT
    MAX(effectiveDate) 
  FROM
    RateHeader 
  WHERE effectiveDate <= :effectiveDate 
    AND rateTableType = :rateTableType 
  ) 
  AND 
  (
    rcolRange.rateColumnType = :colRangeType 
    AND rcRange.rateColumnID = rcolRange.rateColumnID 
    AND rcolRange.rateHeaderID = RateHeader.rateHeaderID 
  ) 
  AND 
  (
    RateRow.rateRowType = :rateRowType 
    AND rcRange.rateRowID = RateRow.rateRowID 
    AND RateRow.rateHeaderID = RateHeader.rateHeaderID 
  ) 
  AND 
  (
    rcolRange.rateColumnKeyMin <= :value 
    AND rcolRange.rateColumnKeyMax >= :value 
  ) 
  AND 
  (
    rCol.rateColumnType = :rateColumnType 
    AND rcValue.rateColumnID = rCol.rateColumnID 
    AND rCol.rateHeaderID = RateHeader.rateHeaderID 
  ) 
  AND 
  (
    rcRange.rateCellIndex = rcValue.rateCellIndex 
  )

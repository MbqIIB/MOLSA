Database SQL Operation: EvidenceDescriptor.searchInEditEvdObjInstanceChanges

Operation Details
Description: Lists all the changes on an active evidence instance.
Sterotype: nsmulti

SQL 
SELECT CASEID, MAX(CHANGEDATETIME), CHANGEUSER, EFFECTIVEFROM, EVIDENCECHANGEHISTORY.EVIDENCEDESCRIPTORID, EVIDENCETYPE, NEWIND, PARTICIPANTID, PENDINGREMOVALIND, PENDINGUPDATEIND, RECEIVEDDATE, RELATEDID, STATUSCODE, SUCCESSIONID, VERSIONNO, CHANGEREASON, EVIDENCEACTIVATIONDATE, CORRECTIONSETID        
INTO :caseID, :changeDateTime, :changeUser, :effectiveFrom, :evidenceDescriptorID, :evidenceType, :newInd, :participantID, :pendingRemovalInd, :pendingUpdateInd, :receivedDate, :relatedID, :statusCode, :successionID, :versionNo, :changeReason, :evidenceActivationDate, :correctionSetID 
FROM EVIDENCEDESCRIPTOR, EVIDENCECHANGEHISTORY 
WHERE SUCCESSIONID = :successionID 
AND STATUSCODE = 'EDS2' 
AND PARTICIPANTID = participantID 
AND EVIDENCECHANGEHISTORY.EVIDENCEDESCRIPTORID = EVIDENCEDESCRIPTOR.EVIDENCEDESCRIPTORID 
AND CASEID = :caseID 
AND EVIDENCETYPE = :evidenceType 
GROUP BY RELATEDID, CHANGEUSER, CHANGEREASON, CORRECTIONSETID, CASEID,  EFFECTIVEFROM, EVIDENCECHANGEHISTORY.EVIDENCEDESCRIPTORID, EVIDENCETYPE, NEWIND, PARTICIPANTID, PENDINGREMOVALIND, PENDINGUPDATEIND, RECEIVEDDATE, STATUSCODE, SUCCESSIONID, VERSIONNO, EVIDENCEACTIVATIONDATE 
ORDER BY EFFECTIVEFROM;

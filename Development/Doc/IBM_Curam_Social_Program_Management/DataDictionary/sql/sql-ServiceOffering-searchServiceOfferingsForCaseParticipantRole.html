Database SQL Operation: ServiceOffering.searchServiceOfferingsForCaseParticipantRole

Operation Details
Description: Finds all active service offerings for the given case participant role that have not ended on or before the date specified.
Sterotype: nsmulti

SQL 
SELECT DISTINCT 
     ServiceOffering.serviceOfferingID,
     ServiceOffering.name,
     ServiceOffering.description,
     ServiceOffering.startDate,
     ServiceOffering.endDate,
     ServiceOffering.unitOfMeasure,
     ServiceOffering.maximumUnits,
     ServiceOffering.unitFrequency,
     ServiceOffering.comments,
     ServiceOffering.recordStatus,
     ServiceOffering.nameUpper,
     ServiceOffering.placementPmtInd,
     ServiceOffering.saPlmtInhibitInd,
     ServiceOffering.payBasedOnAttendanceInd,
     ServiceOffering.trainingInd,
     ServiceOffering.specialConditionType,
     ServiceOffering.versionNo 

     INTO 
     :serviceOfferingID,
     :name,
     :description,
     :startDate,
     :endDate,
     :unitOfMeasure,
     :maximumUnits,
     :unitFrequency,
     :comments,
     :recordStatus,
     :nameUpper,
     :placementPmtInd,
     :saPlmtInhibitInd,
     :payBasedOnAttendanceInd,
     :trainingInd,
     :specialConditionType,
     :versionNo
 
FROM 
   ServiceOffering,ServiceAuthorization,ServiceAuthorizationLineItem 

 WHERE 

 ServiceAuthorization.caseParticipantRoleID=:caseParticipantRoleID
AND ServiceAuthorization.serviceAuthorizationID = ServiceAuthorizationLineItem.serviceAuthorizationID

AND ServiceOffering.serviceOfferingID = ServiceAuthorizationLineItem.serviceID

AND ServiceOffering.recordStatus = :status 

AND (ServiceOffering.endDate is null or ServiceOffering.endDate > :fromDate)

Database SQL Operation: Activity.searchActivityByCaseAndConcern

Operation Details
Description: Reads all the activities By CaseID and ConcernRoleID. Includes the caseReference.
Sterotype: nsmulti

SQL 
SELECT
  Activity.activityID,
  Activity.userName,
  Activity.organisationID,
  Activity.subject,
  Activity.activityTypeCode,
  Activity.startDateTime,
  Activity.endDateTime,
  Activity.allDayInd,
  Activity.locationID,
  Activity.locationName,
  Activity.priorityCode,
  Activity.timeStatusCode,
  Activity.notes,
  Activity.recurrenceID,
  Activity.caseID,
  Activity.concernRoleID,
  Activity.recordStatusCode,
  Activity.versionNo,
  CaseHeader.caseReference
INTO
  :activityID,
  :userName,
  :organisationID,
  :subject,
  :activityTypeCode,
  :startDateTime,
  :endDateTime,
  :allDayInd,
  :locationID,
  :locationName,
  :priorityCode,
  :timeStatusCode,
  :notes,
  :recurrenceID,
  :caseID,
  :concernRoleID,
  :recordStatusCode,
  :versionNo,
  :caseReference 
FROM
  Activity,
  CaseHeader 
WHERE 
  (
    
    (
      CaseHeader.integratedCaseID = :caseID 
      AND CaseHeader.concernRoleID = :concernRoleID 
      AND Activity.caseID = CaseHeader.caseID 
    ) 
    OR 
    (
      CaseHeader.caseID = :caseID 
      AND Activity.caseID = :caseID 
      AND Activity.concernRoleID = :concernRoleID 
    ) 
  )

Database SQL Operation: DeterminationDecision.readDecisionForDisplay

Operation Details
Description:  
Sterotype: ns

SQL 
SELECT
  DeterminationDecision.determinationDecisionID,
  DeterminationDecision.determinationExecutionID,
  DeterminationDecision.resultType,
  DeterminationDecision.status,
  DeterminationDecision.totalScore,
  DeterminationDecision.appliedInd,
  DeterminationDecision.contradictionMessageID,
  DeterminationExecution.decisionMadeBy,
  DeterminationExecution.decisionDateTime,
  DecisionMatrixVersion.type,
  DeterminationDelivery.effectiveDate,
  DeterminationConfig.type,
  DeterminationDecision.decisionMatrixVersionID
INTO
  :determinationDecisionID,
  :determinationExecutionID,
  :resultType,
  :status,
  :totalScore,
  :appliedInd,
  :contradictionMessageID,
  :decisionMadeBy,
  :decisionDateTime,
  :decisionType,
  :effectiveDate,
  :determinationConfigType,
  :decisionMatrixVersionID 
FROM
  DeterminationExecution,
  DeterminationDecision,
  DecisionMatrixVersion,
  DeterminationDelivery,
  DeterminationConfig 
WHERE DeterminationDecision.determinationDecisionID = :determinationDecisionID 
  AND DeterminationExecution.determinationExecutionID = DeterminationDecision.determinationExecutionID 
  AND DecisionMatrixVersion.decisionMatrixVersionID = DeterminationDecision.decisionMatrixVersionID 
  AND DeterminationDelivery.determinationDeliveryID = DeterminationExecution.determinationDeliveryID 
  AND DeterminationDelivery.determinationConfigID = DeterminationConfig.determinationConfigID

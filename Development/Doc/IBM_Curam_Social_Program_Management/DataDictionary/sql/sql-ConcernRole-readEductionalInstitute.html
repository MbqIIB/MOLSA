Database SQL Operation: ConcernRole.readEductionalInstitute

Operation Details
Description: Reads Educational Institute tab details.
Sterotype: ns

SQL 
SELECT
  ConcernRole.concernRoleName,
  EducationalInstitute.schoolType,
  EducationalInstitute.publicPrivateType,
  ConcernRole.primaryAlternateID,
  ConcernRoleAlternateID.typeCode,
  Address.addressData,
  WebAddress.webAddress,
  PhoneNumber.phoneNumberID,
  EmailAddress.emailAddress
INTO
  :name,
  :educationalInstituteType,
  :publicPrivateType,
  :primaryAlternateID,
  :primaryAlternateIDType,
  :addressData,
  :webAddress,
  :phoneNumberID,
  :emailAddress 
FROM
  ConcernRole 
  
  LEFT OUTER JOIN
  
  ConcernRoleAlternateID 
  ON (ConcernRole.primaryAlternateID = ConcernRoleAlternateID.alternateID) 
  
  LEFT OUTER JOIN
  
  Address 
  ON (ConcernRole.primaryAddressID = Address.addressID) 
  
  LEFT OUTER JOIN
  
  WebAddress 
  ON (ConcernRole.primaryWebAddressID = WebAddress.webAddressID) 
  
  LEFT OUTER JOIN
  
  PhoneNumber 
  ON (ConcernRole.primaryPhoneNumberID = PhoneNumber.phoneNumberID) 
  
  LEFT OUTER JOIN
  
  EmailAddress 
  ON (ConcernRole.primaryEmailAddressID = EmailAddress.emailAddressID) 
  
  LEFT OUTER JOIN
  
  EducationalInstitute 
  ON (ConcernRole.concernRoleID = EducationalInstitute.concernRoleID) 
WHERE ConcernRole.concernRoleID = :concernRoleID

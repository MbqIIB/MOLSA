Database SQL Operation: OrganisationStructure.readOrganisationStructureTabDetails

Operation Details
Description: Reads organization structure tab display details.
Sterotype: ns

SQL 
SELECT
  OrganisationStructure.name,
  OrganisationStructure.statusCode,
  OrganisationStructure.activationDate,
  RootOrganisationUnit.organisationUnitID,
  RootOrganisationUnit.name
INTO
  :name,
  :statusCode,
  :activationDate,
  :rootOrgUnitID,
  :rootOrgUnitName
FROM
  OrganisationStructure 
LEFT OUTER JOIN (
  SELECT
    OrganisationUnit.organisationUnitID,
    OrganisationUnit.name,
    OrgUnitParentLink.organisationStructureID
  FROM
    OrganisationUnit,
    OrgUnitParentLink 
  WHERE OrgUnitParentLink.organisationStructureID = :organisationStructureID 
    AND OrgUnitParentLink.parentOrganisationUnitID IS NULL 
    AND OrganisationUnit.organisationUnitID = OrgUnitParentLink.organisationUnitID 
    AND OrgUnitParentLink.recordStatus = :recordStatus) RootOrganisationUnit ON RootOrganisationUnit.organisationStructureID = OrganisationStructure.organisationStructureID
WHERE OrganisationStructure.organisationStructureID = :organisationStructureID

Database SQL Operation: IssueDelivery.readIssueTabDetails

Operation Details
Description:  
Sterotype: ns

SQL 
SELECT 
IssueDelivery.caseID, 
IssueDelivery.issueType,
relatedCaseHeader.caseID,
relatedCaseHeader.caseReference,
relatedCaseHeader.integratedCaseType,
caseHeader1.caseReference,
caseHeader1.statusCode,
caseHeader1.ownerOrgObjectLinkID,
caseHeader1.concernRoleID,
caseHeader1.startDate,
caseHeader1.endDate,
caseHeader1.priorityCode,
OrgObjectLink.orgObjectReference, 
OrgObjectLink.userName, 
OrgObjectLink.orgObjectType,
ConcernRole.concernRoleName,
ConcernRole.concernRoleType,
ProductDelivery.productType

INTO
:caseID, 
:issueType,
:relatedCaseID,
:relatedCaseReference,
:relatedCaseType,
:caseReference,
:caseStatusCode,
:orgObjectLinkID,
:concernRoleID,
:startDate,
:endDate,
:priorityCode,
:orgObjectReference, 
:userName, 
:orgObjectType,
:concernRoleName,
:concernRoleType,
:productType

FROM 
IssueDelivery,
CaseHeader caseHeader1,
OrgObjectLink,
ConcernRole,
CaseHeader relatedCaseHeader
LEFT OUTER JOIN 
ProductDelivery ON (ProductDelivery.caseID = relatedCaseHeader.caseID)

WHERE
IssueDelivery.caseID = :caseID
AND
IssueDelivery.caseID = caseHeader1.caseID
AND 
OrgObjectLink.orgObjectLinkID = caseHeader1.ownerOrgObjectLinkID
AND
IssueDelivery.relatedCaseID = relatedCaseHeader.caseID
AND
ConcernRole.concernRoleID = caseHeader1.concernRoleID

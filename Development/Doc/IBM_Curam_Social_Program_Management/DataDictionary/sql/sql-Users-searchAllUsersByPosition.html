Database SQL Operation: Users.searchAllUsersByPosition

Operation Details
Description: Method to return both active and cancelled users for specified position
Sterotype: nsmulti

SQL 
SELECT
  u.userName,
  u.fullName,
  phl.recordStatus,
  ou.organisationUnitID,
  ou.name,
  pos.positionID,
  pos.name,
  phl.fromDate,
  phl.toDate,
  phl.versionNo
INTO
  :userName,
  :userFullName,
  :positionLinkStatus,
  :organisationUnitID,
  :organisationUnitName,
  :positionID,
  :positionName,
  :fromDate,
  :toDate,
  :versionNo 
FROM
  Position pos,
  OrganisationUnit ou,
  PositionHolderLink phl,
  OrgUnitPositionLink oupl,
  Users u 
WHERE 
  (
    oupl.organisationStructureID = :organisationStructureID 
  ) 
  AND oupl.positionID = :positionID 
  AND pos.positionID = oupl.positionID 
  AND ou.organisationUnitID = oupl.organisationUnitID 
  AND phl.positionID = pos.positionID 
  AND 
  (
    phl.organisationStructureID = :organisationStructureID 
  ) 
  AND u.userName = phl.userName 
  AND phl.fromDate <= :effectiveDate 
  AND 
  (
    phl.toDate IS NULL 
    OR phl.toDate >= :effectiveDate 
  )

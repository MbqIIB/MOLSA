Database SQL Operation: InstructionLineItem.searchByPayslipInstructionID

Operation Details
Description: Operation used to retrieve data using the PayslipInstructionID from the Instruction Line item.
Sterotype: nsmulti

SQL 
SELECT
  InstructionLineItem.instructLineItemID,
  InstructionLineItem.instructionLineItemType,
  InstructionLineItem.amount,
  InstructionLineItem.financialCompID,
  InstructionLineItem.caseID,
  InstructionLineItem.coverPeriodFrom,
  InstructionLineItem.coverPeriodTo,
  InstructionLineItem.effectiveDate,
  InstructionLineItem.concernRoleID,
  InstructionLineItem.primaryClientID,
  InstructionLineItem.caseNomineeID,
  InstructionLineItem.currencyTypeCode,
  PayslipComponent.recipientTypeCode,
  Payslip.recipientName,
  CaseHeader.caseReference
INTO
  :instructionLineItemID,
  :instructionLineItemType,
  :amount,
  :financialComponentID,
  :caseID,
  :coverPeriodFrom,
  :coverPeriodTo,
  :effectiveDate,
  :concernRoleID,
  :primaryClientID,
  :caseNomineeID,
  :currencyTypeCode,
  :recipientTypeCode,
  :recipientName,
  :caseReference 
FROM
  Payslip,
  PayslipComponent,
  InstructionLineItem 
  
  LEFT OUTER JOIN
  
  CaseHeader 
  ON InstructionLineItem.caseID = CaseHeader.caseID 
WHERE Payslip.paySlipInstructionID = :payslipInstructionID 
  AND PayslipComponent.pslipInstructionID = Payslip.paySlipInstructionID 
  AND PayslipComponent.instructLineItemID = InstructionLineItem.instructLineItemID 
ORDER BY caseID,
  coverPeriodFrom ASC

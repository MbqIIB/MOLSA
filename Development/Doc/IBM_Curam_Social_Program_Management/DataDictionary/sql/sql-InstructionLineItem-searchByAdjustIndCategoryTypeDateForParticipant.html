Database SQL Operation: InstructionLineItem.searchByAdjustIndCategoryTypeDateForParticipant

Operation Details
Description: To get a list of instructionLineItem for a particular categoryCode, TypeCode, effectiveDate, and adjustmentInd.
Sterotype: nsmulti

SQL 
SELECT
  ILI.instructLineItemID,
  ILI.amount,
  ILI.unprocessedAmount,
  ILI.instructLineItemCategory,
  ILI.instructionLineItemType,
  ILI.financialCompID,
  ILI.coverPeriodFrom,
  ILI.coverPeriodTo,
  ILI.effectiveDate,
  ILI.creationDate,
  ILI.statusCode,
  ILI.caseID,
  ILI.finInstructionID,
  ILI.concernRoleID,
  ILI.caseNomineeID,
  ILI.primaryClientID,
  ILI.deliveryMethodType,
  ILI.creditDebitType,
  ILI.currencyExchangeID,
  ILI.currencyTypeCode,
  ILI.adjustmentInd,
  ILI.adjustmentFrequency,
  ILI.nextAdjustmentDate,
  ILI.instrumentGenInd,
  ILI.fundID,
  ILI.maximumAmount,
  ILI.versionNo,
  ILI.dueDate,
  ILI.inRespectOfID
INTO
  :instructLineItemID,
  :amount,
  :unprocessedAmount,
  :instructLineItemCategory,
  :instructionLineItemType,
  :financialCompID,
  :coverPeriodFrom,
  :coverPeriodTo,
  :effectiveDate,
  :creationDate,
  :statusCode,
  :caseID,
  :finInstructionID,
  :concernRoleID,
  :caseNomineeID,
  :primaryClientID,
  :deliveryMethodType,
  :creditDebitType,
  :currencyExchangeID,
  :currencyTypeCode,
  :adjustmentInd,
  :adjustmentFrequency,
  :nextAdjustmentDate,
  :instrumentGenInd,
  :fundID,
  :maximumAmount,
  :versionNo,
  :dueDate,
  :inRespectOfID 
FROM
  InstructionLineItem ILI,
  CaseNominee CN,
  CaseParticipantRole CPR 
WHERE CPR.participantRoleID = :participantRoleID 
  AND CN.caseParticipantRoleID = CPR.caseParticipantRoleID 
  AND ILI.caseNomineeID = CN.caseNomineeID 
  AND ILI.instructLineItemCategory = :instructLineItemCategory 
  AND ILI.instructionLineItemType = :instructionLineItemType 
  AND ILI.adjustmentInd = :adjustmentInd 
  AND ILI.effectiveDate <= :effectiveDate

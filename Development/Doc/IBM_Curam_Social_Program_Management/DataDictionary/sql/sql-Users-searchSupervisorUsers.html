Database SQL Operation: Users.searchSupervisorUsers

Operation Details
Description:  
Sterotype: nsmulti

SQL 
SELECT
  distinct Users.userName,
  Users.fullName,
  Position.name,
  Users.surname
INTO
  :userName,
  :userFullName,
  :positionName,
  :surName 
FROM
  Position,
  PositionHolderLink,
  Users 
WHERE Position.positionID = PositionHolderLink.positionID 
  AND PositionHolderLink.userName = Users.userName 
  AND Position.recordStatus <> :status 
  AND PositionHolderLink.recordStatus <> :status 
  AND Position.positionID IN 
  ( 
  SELECT
    OrgUnitPositionLink.positionID 
  FROM
    OrgUnitPositionLink 
  WHERE OrgUnitPositionLink.recordStatus <> :status 
    AND OrgUnitPositionLink.organisationUnitID IN 
    ( 
    SELECT
      opl1.organisationUnitID 
    FROM
      OrgUnitPositionLink opl1,
      PositionHolderLink phl2,
      Position p1 
    WHERE p1.positionID = phl2.positionID 
      AND p1.leadPositionInd = '1' 
      AND phl2.positionID = opl1.positionID 
      AND phl2.userName = :userName 
      AND opl1.recordStatus <> :status 
      AND phl2.recordStatus <> :status 
    
    UNION
      
    SELECT
      opar.organisationUnitID 
    FROM
      OrgUnitParentLink opar 
    WHERE opar.recordStatus <> :status 
      AND opar.parentOrganisationUnitID IN 
      ( 
      SELECT
        opl3.organisationUnitID 
      FROM
        OrgUnitPositionLink opl3,
        PositionHolderLink phl3,
        Position position1 
      WHERE position1.positionID = phl3.positionID 
        AND phl3.positionID = opl3.positionID 
        AND phl3.userName = :userName 
        AND position1.leadPositionInd = '1' 
        AND phl3.recordStatus <> :status 
        AND opl3.recordStatus <> :status 
      ) 
    ) 
  
  UNION
    
  SELECT
    prl.positionID 
  FROM
    PositionReportingLink prl,
    PositionHolderLink phl4,
    Position position2 
  WHERE phl4.positionID = prl.reportsToID 
    AND position2.leadPositionInd = '1' 
    AND phl4.userName = :userName 
    AND phl4.recordStatus <> :status 
    AND prl.recordStatus <> :status 
  ) 
ORDER BY Users.surname

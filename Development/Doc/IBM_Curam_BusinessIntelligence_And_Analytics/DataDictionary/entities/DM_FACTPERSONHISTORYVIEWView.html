SQL 
select
   T1.DWPARTICIPANTSTATUSID as DWPARTICIPANTSTATUSID,
   T1.GENDER as GENDER,
   T1.MARITALSTATUS as MARITALSTATUS,
   T1.COUNTRYOFBIRTH as COUNTRYOFBIRTH,
   T1.DATEOFBIRTH as DATEOFBIRTH,
   T1.STARTDATE as STARTDATE,
   T1.LASTWRITTEN as LASTWRITTEN,
   T3.PREFERREDLANGUAGE as PREFERREDLANGUAGE,
   t3.REGISTRATIONDATE as REGDATE,
   t3.DWADDRESSID as DWADDRESSID,
   T1.ETHNICORIGIN as ETHNICORIGIN,
   T1.DWPERSONHISTORYID as DWPERSONHISTORYID,
   T3.CONCERNROLEID as CONCERNROLEID,
   t3.SSN as SSN,
   T1.FIRSTNAME as FIRSTNAME,
   T1.LASTNAME as LASTNAME,
   T1.DWPERSONTYPEKEY as DWPERSONTYPEKEY,
   T1.DWNATIONALITYKEY as DWNATIONALITYKEY,
   T1.DWINDIGENOUSGROUPKEY as DWINDIGENOUSGROUPKEY,
   T1.DWRACETYPEKEY as DWRACETYPEKEY,
   T1.DWCONSTSTATKEY as DWCONSTSTATKEY 
from
   DW_PERSONHISTORY t1,
   DW_PARTICIPANT T3
where
   T1.DWPARTICIPANTID = T3.DWPARTICIPANTID;
